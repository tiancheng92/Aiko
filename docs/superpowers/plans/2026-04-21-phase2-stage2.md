## Task 2：后端新增 GetAvailableModels 方法

**Files:**
- Modify: `app.go`

`frontend/public/live2d/` 目录下每个子目录（排除 `core`）即为一个可用模型，目录名即模型名。

- [ ] **Step 1: 在 app.go 末尾添加方法**

```go
// GetAvailableModels returns a list of available Live2D model names by
// scanning subdirectories of the bundled live2d assets directory.
// The special "core" directory is excluded.
func (a *App) GetAvailableModels() []string {
	// Wails serves static files from frontend/public; at runtime the
	// WebView root maps to that directory, but the Go process needs the
	// filesystem path. We locate it relative to the executable's working dir.
	base := "frontend/public/live2d"
	entries, err := os.ReadDir(base)
	if err != nil {
		return []string{"hiyori"}
	}
	var models []string
	for _, e := range entries {
		if e.IsDir() && e.Name() != "core" {
			models = append(models, e.Name())
		}
	}
	if len(models) == 0 {
		return []string{"hiyori"}
	}
	return models
}
```

## Task 3：前端 useModelPath composable

**Files:**
- Create: `frontend/src/composables/useModelPath.js`

composable 职责：监听后端 `config:updated` 事件 / 初始加载配置，返回响应式 `modelPath`（格式：`/live2d/<modelName>/<ModelName>.model3.json`，首字母大写）。

- [ ] **Step 1: 创建文件**

```js
// frontend/src/composables/useModelPath.js
import { ref, onMounted, onUnmounted } from 'vue'
import { GetConfig } from '../../wailsjs/go/main/App'
import { EventsOn } from '../../wailsjs/runtime/runtime'

/**
 * buildModelPath converts a model directory name to the full model JSON path.
 * Convention: directory "hiyori" → "/live2d/hiyori/Hiyori.model3.json"
 */
function buildModelPath(name) {
  if (!name) name = 'hiyori'
  const capitalized = name.charAt(0).toUpperCase() + name.slice(1)
  return `/live2d/${name}/${capitalized}.model3.json`
}

/**
 * useModelPath provides a reactive Live2D model path that updates
 * whenever the saved configuration changes.
 */
export function useModelPath() {
  const modelPath = ref(buildModelPath('hiyori'))
  let offUpdate = null

  onMounted(async () => {
    try {
      const cfg = await GetConfig()
      if (cfg?.Live2DModel) modelPath.value = buildModelPath(cfg.Live2DModel)
    } catch {}
    offUpdate = EventsOn('config:updated', (modelName) => {
      if (modelName) modelPath.value = buildModelPath(modelName)
    })
  })

  onUnmounted(() => offUpdate?.())

  return { modelPath }
}
```

/**
 * watchModelPath reloads the Live2D model when the configured model changes.
 * The PixiJS Application is reused; only the model on stage is replaced.
 */
watch(modelPath, async (newPath) => {
  if (!pixiApp) return
  if (live2dModel) {
    live2dModel.off('hit')
    pixiApp.stage.removeChild(live2dModel)
    live2dModel.destroy()
    live2dModel = null
  }
  try {
    const newModel = await Live2DModel.from(newPath, { autoInteract: false })
    if (!mounted) { newModel.destroy(); return }
    live2dModel = newModel
    pixiApp.stage.addChild(live2dModel)
    const scale = petSize.value / live2dModel.internalModel.originalHeight
    live2dModel.scale.set(scale)
    live2dModel.anchor.set(0.5, 0.5)
    live2dModel.position.set(petSize.value / 2, petSize.value / 2)
    live2dModel.on('hit', (hitAreas) => {
      if (hitAreas.includes('Body')) {
        live2dModel.motion('TapBody', undefined, MotionPriority.NORMAL)
      }
    })
    live2dModel.motion('Idle', undefined, MotionPriority.IDLE)
  } catch (err) {
    console.error('Live2DPet model reload failed:', err)
  }
})
```

const copiedIdx = ref(-1)

/** copyMessage copies message content to the clipboard and briefly shows a checkmark. */
async function copyMessage(content, idx) {
  try {
    await navigator.clipboard.writeText(content)
    copiedIdx.value = idx
    setTimeout(() => { copiedIdx.value = -1 }, 1500)
  } catch {
    // Fallback for environments where clipboard API is unavailable
    const el = document.createElement('textarea')
    el.value = content
    document.body.appendChild(el)
    el.select()
    document.execCommand('copy')
    document.body.removeChild(el)
    copiedIdx.value = idx
    setTimeout(() => { copiedIdx.value = -1 }, 1500)
  }
}
```

  if (messages.value[idx]?.thinking) messages.value.splice(idx, 1)
  messages.value.push({ role: 'system', content: '错误: ' + err })
  loading.value = false
  EventsEmit('pet:state:change', 'error')
})
